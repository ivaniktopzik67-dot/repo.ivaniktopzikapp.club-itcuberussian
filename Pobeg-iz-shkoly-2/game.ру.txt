import time

def print_slow(text):
    for char in text:
        print(char, end='', flush=True)
        time.sleep(0.03)
    print()

def show_intro():
    print_slow("ПОБЕГ ИЗ ШКОЛЫ 2: СБЕЖАТЬ ИЗ УЧИЛИЩА\n")
    print_slow("Игрок: Тот\n")
    print_slow("\nТы — Тот, студент Училища — места, где свобода — лишь слово.\n")
    print_slow("Здесь всё по расписанию, камеры на каждом углу, а за любой проступок — дополнительные отработки.\n")
    print_slow("Сегодня ты решишь бежать. Сейчас или никогда.\n")

def chapter_1():
    print_slow("\nГЛАВА 1. ПОДГОТОВКА\n")
    print_slow("Ты в своей комнате в общежитии. До вечерней проверки осталось 2 часа.\n")
    print("1. Осмотреть комнату — поискать полезные предметы.")
    print("2. Выйти в коридор и разведать обстановку.")
    print("3. Подождать до темноты.")
    choice = input("\nТвой выбор (1/2/3): ")
    if choice == "1":
        print_slow("Ты находишь перочинный нож, моток верёвки и старую карту территории.\n")
        return {"knife": True, "rope": True, "map": True}
    elif choice == "2":
        print_slow("В коридоре тихо, но у лестницы стоит охранник. Ты возвращаешься в комнату.\n")
        return {}
    else:
        print_slow("Время уходит. Патрули усиливаются. Выбраться будет сложнее.\n")
        return {"delayed": True}

def chapter_2(inventory):
    print_slow("\nГЛАВА 2. ПУТЬ ЧЕРЕЗ УЧИЛИЩЕ\n")
    print("1. Через спортзал — там обычно никого нет, но дверь может быть заперта.")
    print("2. По главному коридору — быстро, но много камер.")
    print("3. Через подвал — темно и сыро, зато мало кто туда ходит.")
    choice = input("\nТвой выбор (1/2/3): ")
    if choice == "1" and inventory.get("knife"):
        print_slow("Дверь заперта. Ты используешь нож, чтобы поддеть замок. Получилось!\n")
        return True
    elif choice == "2":
        print_slow("Камеры фиксируют твоё движение. Раздаётся сигнал тревоги!\n")
        print_slow("Охранники бегут к тебе. Нужно прятаться!\n")
        return False
    elif choice == "3":
        print_slow("В подвале сыро и темно, но ты находишь старый тоннель к внешней стене.\n")
        return True
    else:
        print_slow("Ты застрял. Охранники приближаются!\n")
        return False

def chapter_3():
    print_slow("\nГЛАВА 3. ТЕРРИТОРИЯ УЧИЛИЩА\n")
    print("До забора — 50 метров открытого пространства. Вокруг патрули.\n")
    print("1. Бежать напрямую — быстро, но заметят.")
    print("2. Прятаться за кустами и продвигаться медленно.")
    print("3. Использовать верёвку, чтобы залезть на дерево и перебраться сверху.")
    choice = input("\nТвой выбор (1/2/3): ")
    if choice == "1":
        print_slow("Охранники замечают тебя сразу. Они бегут за тобой!\n")
        return "run"
    elif choice == "2":
        print_slow("Ты медленно продвигаешься, прячась за кустами. Охранники проходят мимо.\n")
        return "hide"
    else:
        print_slow("Дерево скрипит под весом. Один охранник поднимает голову: «Эй, там кто‑то!»\n")
        return "tree"

def final_jump(path, inventory):
    print_slow("\nФИНАЛ. ПРЫЖОК ЧЕРЕЗ ЗАБОР\n")
    print("Высокий забор с колючей проволокой. Вокруг — охранники.\n")
    print("1. Прыгнуть прямо сейчас, пока они не подошли близко.")
    print("2. Бросить в них верёвку и нож, чтобы отвлечь.")
    print("3. Заговорить с ними, убедить, что ты просто вышел прогуляться.")
    choice = input("\nТвой выбор (1/2/3): ")
    if choice == "1" and path == "hide":
        print_slow("\nТы разбегаешься и прыгаешь! Цепляешься за верх, подтягиваешься и — свобода!\n")
        print_slow("Охранники подбегают, но уже поздно. Ты свободен!\n")
        return "good"
    elif choice == "2" and inventory.get("rope") and inventory.get("knife"):
        print_slow("\nОхранники отвлекаются на брошенные предметы. Ты перелезаешь через забор!\n")
        print_slow("Они понимают, что их обманули, но ты уже далеко.\n")
        return "good"
    elif choice == "3":
        print_slow("\nОхранники не верят тебе. «Стой на месте!» — кричат они.\n")
        print_slow("Один хватает тебя за руку. Попытка побега провалилась.\n")
        return "bad"
    else:
        print_slow("\nСлишком поздно! Охранники окружают тебя. Попытка побега неудачна.\n")
        return "bad"

def show_ending(ending):
    print_slow("\nКОНЕЦ ИГРЫ\n")
    if ending == "good":
        print_slow("Хорошая концовка: Ты бежишь по полю, ветер свистит в ушах. Свобода!\n")
        print_slow("Ты оставил Училище позади. Теперь можно начать новую жизнь.\n")
    elif ending == "bad":
        print_slow("Плохая концовка: Тебя ведут обратно в Училище.\n")
        print_slow("Наказание будет строгим: месяц без прогулок и строгий надзор.\n")
        print_slow("Но ты знаешь, что попробуешь снова.\n")
    else:
        print_slow("Секретная концовка: Ты не просто сбежал — ты забрал карту с тайнами Училища.\n")
        print_slow("Позже ты передаёшь её журналистам, и о нарушениях узнают все.\n")

def main():
    show_intro()
    inventory = chapter_1()
    success = chapter_2(inventory)
    if not success:
        show_ending("bad")
        return
    path = chapter_3()
    ending = final_jump(path, inventory)
    show_ending(ending)

if __name__ == "__main__":
    main()
